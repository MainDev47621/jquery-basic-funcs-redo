Basic jQuery Effects

$('.show-btn').on('click', () => {

    $('.image-one').show();

});


$('.hide-btn').on('click', () => {

    $('.image-one').hide();

});


$('.toggle-btn').on('click', () => {

    $('.image-one').toggle();

});

$('.fade-in-btn').on('click', () => {

    $('.image-two').fadeIn()

});


$('.fade-out-btn').on('click', () => {

    $('.image-two').fadeOut()

});


$('.fade-toggle-btn').on('click', () => {

    $('.image-two').fadeToggle()

});

$('.slide-down-btn').on('click', () => {

    $('.image-one').slideDown();

});


$('.slide-up-btn').on('click', () => {

    $('.image-one').slideUp();

});


$('.slide-toggle-btn').on('click', () => {

    $('.image-one').slideToggle();

});
